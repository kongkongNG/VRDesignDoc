109  public static void PerformCSGCalculate(GameObject lhs, GameObject rhs, CSGType type, bool isShowAreas = false)
110  {
111  // 获取模型
112  if (lhs == null || rhs == null)
113  {
114  Debug.LogError("[DEBUG] lhs or rhs missing in CSGAction");
115  return;
116  }
117  SGModel lhsModel = lhs.GetComponent<SGModel>();
118  SGModel rhsModel = rhs.GetComponent<SGModel>();
119 
120  if (lhsModel == null || rhsModel == null)
121  {
122  Debug.LogError("[DEBUG] lhs_SGModel or rhs_SGModel missing in CSGAction");
123  return;
124  }
125 
126  if (lhsModel.op_mesh == IntPtr.Zero)
127  {
128  Debug.LogError("[DEBUG] lhs op_mesh missing in CSGAction");
129  return;
130  }
131 
132  // 执行布尔运算
133  IntPtr rhsOpmesh = rhsModel.op_mesh;
134  IntPtr sub_group = IntPtr.Zero;
135  switch (type)
136  {
137  case CSGType.SUBTRACT:
138  sub_group = sguSub(lhsModel.op_mesh, rhsOpmesh);
139  break;
140  case CSGType.UNION:
141  sub_group = sguUnion(lhsModel.op_mesh, rhsOpmesh);
142  break;
143  case CSGType.INTERSECT:
144  sub_group = sguIntersection(lhsModel.op_mesh, rhsOpmesh);
145  break;
146  default:
147  break;
148  }
149  if (sub_group == IntPtr.Zero)
150  {
151  Debug.Log("[DEBUG] sub group == NULL, maybe two model not contact with each other");
152  return;
153  }
154  int obj_count = sguGetGroupObjectNum(sub_group);
155 
156  // 创建CSG后的物体
157  for (int i = 0; i < obj_count; i++)
158  {
159  IntPtr chi = sguGetGroupObject(sub_group, i);
160 
161  GameObject root = new GameObject();
162  root.transform.position = lhs.transform.position;
163  SGModel chiModel = root.AddComponent<SGModel>();
164  chiModel.Init(chi, lhsModel.sg_height);
165 
166  chiModel.ShowAreas();
167  chiModel.ShowEdges();
168  chiModel.PostCreation();
169  if (isShowAreas)
170  chiModel.SetAreasSelectable();
171  }
172 
173  // 销毁操作对象
174  Destroy(lhs);
175  Destroy(rhs);
176  }
177 }
