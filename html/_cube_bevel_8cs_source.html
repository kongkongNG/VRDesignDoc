246  private void RoundCube()
247  {
248  //Vector3[] vertices = new Vector3[mesh.vertices.Length];
249  vertices = mesh.vertices;
250  normals = new Vector3[vertices.Length];
251  inners = new Vector3[vertices.Length];
252 
253  //Debug.Log(extens);
254 
255  for (int i = 0; i < verticesRaw.Length; i++)
256  {
257  Vector3 inner = verticesRaw[i];
258  float x = verticesRaw[i].x, y = verticesRaw[i].y, z = verticesRaw[i].z;
259 
260  // 遍历所有已选择的边
261  foreach (CubeEditEdge ce in selectedEdges)
262  {
263  ce.roundAction(ref inner, x, y, z);
264  inners[i] = inner;
265  normals[i] = (verticesRaw[i] - inner).normalized;
266  vertices[i] = inner + normals[i] * roundness;
267  }
268  }
269 
270  mesh.vertices = vertices;
271  mesh.normals = normals;
272  mesh.RecalculateNormals();
273  }
