254  private SwipeDirection CalculateSwipeAction()
255  {
256 
257  float deltaTime = Time.time - touchStartTime;
258 
259  Vector2 swipeVector = touchEndPosition - touchStartPosition;
260  //float velocity = swipeVector.magnitude / deltaTime;
261 
262  //Debug.Log("[DEBUG] delta time: " + deltaTime);
263  //Debug.Log("[DEBUG] touchStartPosition: " + touchStartPosition);
264  //Debug.Log("[DEBUG] touchEndPosition: " + touchEndPosition);
265  //Debug.Log("[DEBUG] swipeVector.magnitude: " + swipeVector.magnitude);
266 
267  // 达到一定的速度和长度阈值，则确定为swipe行为
268  if ((deltaTime > SwipeMinDeltaTime) && (swipeVector.magnitude > SwipeMinDist))
269  {
270  swipeVector.Normalize();
271  // 计算方向向量与touchpad的x轴夹角
272  float angleOfSwipe = Vector2.Dot(swipeVector, touchpadXaxis);
273  angleOfSwipe = Mathf.Acos(angleOfSwipe) * Mathf.Rad2Deg;
274 
275  // Left / right
276  if (angleOfSwipe < AngleTolerance)
277  {
278  return SwipeDirection.SwipeRight;
279  }
280  else if ((180.0f - angleOfSwipe) < AngleTolerance)
281  {
282  return SwipeDirection.SwipeLeft;
283  }
284  else
285  {
286  // 计算方向向量与touchpad的y轴夹角
287  angleOfSwipe = Vector2.Dot(swipeVector, touchpadYaxis);
288  angleOfSwipe = Mathf.Acos(angleOfSwipe) * Mathf.Rad2Deg;
289  // Top / bottom
290  if (angleOfSwipe < AngleTolerance)
291  {
292  return SwipeDirection.SwipeTop;
293  }
294  else if ((180.0f - angleOfSwipe) < AngleTolerance)
295  {
296  return SwipeDirection.SwipeBottom;
297  }
298  }
299  }
300 
301  return SwipeDirection.None;
302  }
