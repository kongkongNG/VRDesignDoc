256  private SwipeDirection CalculateSwipeAction()
257  {
258 
259  float deltaTime = Time.time - touchStartTime;
260 
261  Vector2 swipeVector = touchEndPosition - touchStartPosition;
262 
263  //Debug.Log("[DEBUG] delta time: " + deltaTime);
264  //Debug.Log("[DEBUG] touchStartPosition: " + touchStartPosition);
265  //Debug.Log("[DEBUG] touchEndPosition: " + touchEndPosition);
266  //Debug.Log("[DEBUG] swipeVector.magnitude: " + swipeVector.magnitude);
267 
268  // 达到一定的速度和长度阈值，则确定为swipe行为
269  if ((deltaTime > SwipeMinDeltaTime) && (swipeVector.magnitude > SwipeMinDist))
270  {
271  swipeVector.Normalize();
272  // 计算方向向量与touchpad的x轴夹角
273  float angleOfSwipe = Vector2.Dot(swipeVector, touchpadXaxis);
274  angleOfSwipe = Mathf.Acos(angleOfSwipe) * Mathf.Rad2Deg;
275 
276  // Left / right
277  if (angleOfSwipe < AngleTolerance)
278  {
279  return SwipeDirection.SwipeRight;
280  }
281  else if ((180.0f - angleOfSwipe) < AngleTolerance)
282  {
283  return SwipeDirection.SwipeLeft;
284  }
285  else
286  {
287  // 计算方向向量与touchpad的y轴夹角
288  angleOfSwipe = Vector2.Dot(swipeVector, touchpadYaxis);
289  angleOfSwipe = Mathf.Acos(angleOfSwipe) * Mathf.Rad2Deg;
290  // Top / bottom
291  if (angleOfSwipe < AngleTolerance)
292  {
293  return SwipeDirection.SwipeTop;
294  }
295  else if ((180.0f - angleOfSwipe) < AngleTolerance)
296  {
297  return SwipeDirection.SwipeBottom;
298  }
299  }
300  }
301 
302  return SwipeDirection.None;
303  }
