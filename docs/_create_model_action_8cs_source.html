201  private void UpdateContourCenter()

207  Vector3 origin = headset.transform.position;
208  Vector3 _dir = headset.transform.forward - headset.transform.up * 0.2f; // 视线在头显稍微偏移下方一点
209  float k = -Vector3.Dot(origin - areaCenterPos, areaNormal) / (Vector3.Dot(_dir, areaNormal));
210  Vector3 targetPos = origin + k * _dir;
