228  protected virtual IEnumerator TweenMenuScale(bool show)
229  {
230  float targetScale = 0;
231  Vector3 direction = -1 * Vector3.one;
232  if (show)
233  {
234  canvasObject.transform.localScale = new Vector3(CanvasScaleSize, CanvasScaleSize, CanvasScaleSize);
235  targetScale = zoomScaleMultiplier;
236  direction = Vector3.one;
237  }
238  int i = 0;
239  while (i < 250 && ((show && transform.localScale.x < targetScale) || (!show && transform.localScale.x > targetScale)))
240  {
241  transform.localScale += direction * Time.deltaTime * 4f * zoomScaleMultiplier;
242  yield return true;
243  i++;
244  }
245  transform.localScale = direction * targetScale;
246 
247  if (!show)
248  {
249  canvasObject.transform.localScale = Vector3.zero;

324  protected virtual void CalculateSwipeAction()
325  {
326  isPendingSwipeCheck = false;
327 
328  float deltaTime = Time.time - touchStartTime;
329  Vector2 swipeVector = touchEndPosition - touchStartPosition;
330  float velocity = swipeVector.magnitude / deltaTime;
331 
332  if ((velocity > SwipeMinVelocity) && (swipeVector.magnitude > SwipeMinDist))
333  {
334  swipeVector.Normalize();
335  float angleOfSwipe = Vector2.Dot(swipeVector, xAxis);
336  angleOfSwipe = Mathf.Acos(angleOfSwipe) * Mathf.Rad2Deg;
337 
338  // Left / right
339  if (angleOfSwipe < AngleTolerance)
340  {
341  OnSwipeRight();
342  }
343  else if ((180.0f - angleOfSwipe) < AngleTolerance)
344  {
345  OnSwipeLeft();
346  }
347  else
348  {
349  // Top / bottom
350  angleOfSwipe = Vector2.Dot(swipeVector, yAxis);
351  angleOfSwipe = Mathf.Acos(angleOfSwipe) * Mathf.Rad2Deg;
352  if (angleOfSwipe < AngleTolerance)
353  {
354  OnSwipeTop();
355  }
356  else if ((180.0f - angleOfSwipe) < AngleTolerance)
357  {
358  OnSwipeBottom();
359  }
360  }
361  }
362  }
