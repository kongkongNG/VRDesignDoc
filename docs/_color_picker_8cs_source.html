143  private void ChangeHueSaturation(Vector2 touchpadAxis, float touchpadAngle)
144  {
145  // 映射hue色调
146  float normalAngle = touchpadAngle - 90; // 转换角度，90度位置为0
147  if (normalAngle < 0)

149  normalAngle += 360;

151  hue = normalAngle / 360.0f;
152 
153  // 映射saturation饱和度
154  float rads = normalAngle * Mathf.PI / 180.0f;
155  float maxX = Mathf.Cos(rads);
156  float maxY = Mathf.Sin(rads);
157  float percentX = Mathf.Abs(touchpadAxis.x / maxX);
158  float percentY = Mathf.Abs(touchpadAxis.y / maxY);
159  saturation = (percentX + percentY) / 2;
160  }
